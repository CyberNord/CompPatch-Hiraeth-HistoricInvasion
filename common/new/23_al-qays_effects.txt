# invader appears, before story
spawn_ali_character_effect = {
	title:c_ubulla.title_province = {
		save_scope_as = $INVADER_DYNN$_spawn_location
	}

	# grandmother
	create_character = {
		template = muhammad_al-qays_mother_character_template
		location = scope:$INVADER_DYNN$_spawn_location
		save_scope_as = $INVADER_NAME$_grandmother
	}

	# mother
	create_character = {
		template = hi_random_spouse_character_template
		dynasty = none
		gender = female
		location = scope:$INVADER_DYNN$_spawn_location
		faith = faith:ashari
		culture = culture:bedouin
		save_scope_as = $INVADER_NAME$_mother
		# after_creation = {
		# 	death = natural
		# }
	}

	# father
	create_character = {
		template = muhammad_al-qays_character_template
		location = scope:$INVADER_DYNN$_spawn_location
		mother = scope:$INVADER_NAME$_grandmother
		save_scope_as = $INVADER_NAME$_father
		after_creation = {
			marry = scope:$INVADER_NAME$_mother
			death = natural
			scope:$INVADER_NAME$_mother = { death = natural }
		}
	}

	# INVADER
	create_character = {
		template = ali_al-qays_character_template
		dynasty = inherit
		mother = scope:$INVADER_NAME$_mother
		father = scope:$INVADER_NAME$_father
		location = scope:$INVADER_DYNN$_spawn_location
		save_scope_as = $INVADER_NAME$
	}
	scope:$INVADER_NAME$ = {
		add_gold = 500
		add_prestige = 250
		add_piety = 50

		dynasty = {
			set_dynasty_name = hi_dynn_al-qays
			add_dynasty_prestige_level = 0
			add_dynasty_prestige = 150
		}
		house = {
			set_house_name = hi_dynn_al-qays
		}
	}

	every_player = {
		trigger_event = $INVADER_DYNN$_invasion.1001 # Notification: Start of invasion
	}
}

### populating the life of the invader character
spawn_al-qays_family_effect = {
	# random spouse
	create_character = {
		template = hi_random_spouse_character_template
		dynasty = none
		employer = $INVADER_CHAR$
		faith = $INVADER_CHAR$.faith
		culture = $INVADER_CHAR$.culture
		after_creation = {
			marry = $INVADER_CHAR$
		}
	}
}
