# spawn invader, before story
spawn_haraldr_character_effect = {
	title:c_vingulmork.title_province = {
		save_scope_as = $INVADER_DYNN$_spawn_location
	}

	# father
	character:168323 = { save_scope_as = halfdan }

	if = {
		limit = { NOT = { exists = scope:$INVADER_NAME$ } }
		# INVADER
		create_character = {
			template = haraldr_yngling_character_template
			dynasty = inherit
			father = scope:halfdan
			location = scope:$INVADER_DYNN$_spawn_location
			save_scope_as = $INVADER_NAME$
			after_creation = {
				add_character_flag = spawned_$INVADER_DYNN$
			}
		}
	}

	scope:$INVADER_NAME$ = {
		add_gold = 750
		add_prestige = 100
		add_piety = 50
		
		dynasty = {
			add_dynasty_prestige_level = 1
			add_dynasty_prestige = 500
			add_dynasty_perk = warfare_legacy_1
		}
	}

	every_player = {
		trigger_event = $INVADER_DYNN$_invasion.1001 # Notification: Start of invasion
	}
}

yngling_maintenance_effect = {
	apply_invader_modifiers_effect = {
		INVADER_DYNN = $INVADER_DYNN$
	}
	if = {
		limit = { $INVADER_CHAR$ = { has_character_flag = spawned_$INVADER_DYNN$ } }
		give_invader_land_effect = {
			INVADER_DYNN = $INVADER_DYNN$
			INVADER_CHAR = $INVADER_CHAR$
			STARTING_TIER = $STARTING_TIER$
		}
	}
	spawn_$INVADER_DYNN$_family_effect = {
		INVADER_CHAR = $INVADER_CHAR$
	}
	spawn_invader_servants_effect = {
		INVADER_CHAR = $INVADER_CHAR$
	}
	form_the_invader_title_effect = {
		INVADER_DYNN = $INVADER_DYNN$
		INVADER_CHAR = $INVADER_CHAR$
		INVADER_TITLE = $INVADER_TITLE$
		LANDLESS_YES_NO = $LANDLESS_YES_NO$
		INVADER_TITLE_TIER = $INVADER_TITLE_TIER$
	}
}

### populating the life of the invader character
spawn_yngling_family_effect = {
	# spouse
	if = {
		limit = { $INVADER_CHAR$ = { has_character_flag = spawned_yngling } }
		create_character = {
			age = { 18 22 }
			template = hi_random_spouse_character_template
			name = rognhildr_character_template_name
			employer = $INVADER_CHAR$
			dynasty = none
			save_scope_as = rognhildr
			after_creation = {
				disease_immunity_short_effect = yes
				set_relation_soulmate = $INVADER_CHAR$
				marry = $INVADER_CHAR$
			}
		}
	}
}
