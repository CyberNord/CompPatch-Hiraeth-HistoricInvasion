# spawn invader, before story
spawn_chandragupta_character_effect = {
	title:c_magadha.title_province = {
		save_scope_as = $INVADER_DYNN$_spawn_location
	}

	# INVADER
	create_character = {
		template = chandragupta_maurya_character_template
		dynasty = generate
		location = scope:$INVADER_DYNN$_spawn_location
		save_scope_as = $INVADER_NAME$
	}

	scope:$INVADER_NAME$ = {
		add_gold = 5000
		add_prestige = 7500
		add_piety = 250
		
		dynasty = {
			set_dynasty_name = hi_dynn_maurya
			set_coa = title:e_maurya

			add_dynasty_prestige_level = 2
			add_dynasty_prestige = 3000
			add_dynasty_perk = hth_warfare_legacy_1
		}
		house = {
			set_house_name = hi_dynn_maurya
		}
	}

	every_player = {
		trigger_event = $INVADER_DYNN$_invasion.1001 # Notification: Start of invasion
	}
}
