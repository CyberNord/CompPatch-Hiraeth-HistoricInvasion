# spawn invader, before story
spawn_yahya_character_effect = {
	title:c_wadan.title_province = { # d_adrar
		save_scope_as = $INVADER_DYNN$_spawn_location
	}

	### add direct line to invader before they spawn themselves so the heritage line can work properly

	# GEN A

	create_character = {
		template = turgurt_almoravid_a_character_template
		name = "Turgurt"
		location = scope:$INVADER_DYNN$_spawn_location
		save_scope_as = turgurt_almoravid_a
	}

	# GEN B

	create_character = {
		template = talagagin_almoravid_b_character_template
		dynasty = inherit
		location = scope:$INVADER_DYNN$_spawn_location
		father = scope:turgurt_almoravid_a
		save_scope_as = talagagin_almoravid_b
	}

	# GEN C

	create_character = {
		template = umar_almoravid_c_character_template
		dynasty = inherit
		location = scope:$INVADER_DYNN$_spawn_location
		father = scope:talagagin_almoravid_b
		save_scope_as = umar_almoravid_c
	}

	# GEN D

	create_character = { # INVADER
		template = yahya_almoravid_character_template
		dynasty = inherit
		location = scope:$INVADER_DYNN$_spawn_location
		father = scope:umar_almoravid_c
		save_scope_as = $INVADER_NAME$
	}

	scope:$INVADER_NAME$ = {
		add_gold = 4500
		add_prestige = 4000
		add_piety = 250

		dynasty = {
			set_dynasty_name = hi_dynn_almoravid
			set_coa = title:e_almoravid

			add_dynasty_prestige_level = 1
			add_dynasty_prestige = 1000
			add_dynasty_perk = hth_warfare_legacy_1
			add_dynasty_perk = hth_erudition_legacy_1
		}
		house = {
			set_house_name = hi_dynn_almoravid
		}
	}

	every_player = {
		trigger_event = $INVADER_DYNN$_invasion.1001 # Notification: Start of invasion
	}
}
